CREATE TABLE Answer(
		AnswerId SMALLINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
		Answer VARCHAR(255) NOT NULL
)

CREATE TABLE Question(
		QuestionId SMALLINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
		Question VARCHAR(255) NOT NULL
)

CREATE TABLE QuestionAnswer(
		QuestionAnswerId SMALLINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
		FOREIGN KEY (AnswerId) REFERENCES Answer,
		FOREIGN KEY (QuestionId) REFERENCES Question
)

CREATE TABLE Feedback(
		FeedbackId SMALLINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
		FOREIGN KEY(QuestionAnswerId) REFERENCES QuestionAnswer
)
